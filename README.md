  <li>ejs-mate</li>
  <li>express</li>
  <li>express-handlebars</li>
  <li>express-session</li>

  <li>passport</li>
  <li>passport-local</li>
</ul>
<p>Y el siguiente como una dependencia:</p>
<code>npm install "nombre del modulo" -D</code>

</ul>
